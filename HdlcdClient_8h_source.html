291  void OnTcpSocketConnected() {
292  // Checks
293  if ((m_eTcpSocketDataState == SOCKET_STATE_CONNECTING) && (m_eTcpSocketCtrlState == SOCKET_STATE_CONNECTING)) {
294  // Invalid state
295  assert(false);
296  return;
297  } // if
298 
299  if ((m_eTcpSocketDataState == SOCKET_STATE_CONNECTED) && (m_eTcpSocketCtrlState == SOCKET_STATE_CONNECTED)) {
300  // Success!
301  // Create and start the packet endpoint for the exchange of user data packets
302  m_PacketEndpointData = std::make_shared<HdlcdPacketEndpoint>(m_IOService, std::make_shared<FrameEndpoint>(m_IOService, m_TcpSocketData));
303  m_PacketEndpointData->SetOnDataCallback([this](std::shared_ptr<const HdlcdPacketData> a_PacketData){ return OnDataReceived(a_PacketData); });
304  m_PacketEndpointData->SetOnClosedCallback([this](){ OnClosed(); });
305  m_PacketEndpointData->Start();
306  m_PacketEndpointData->Send(HdlcdSessionHeader::Create(m_HdlcdSessionDescriptor, m_SerialPortName));
307 
308  // Create and start the packet endpoint for the exchange of control packets
309  m_PacketEndpointCtrl = std::make_shared<HdlcdPacketEndpoint>(m_IOService, std::make_shared<FrameEndpoint>(m_IOService, m_TcpSocketCtrl));
310  m_PacketEndpointCtrl->SetOnCtrlCallback([this](const HdlcdPacketCtrl& a_PacketCtrl){ return OnCtrlReceived(a_PacketCtrl); });
311  m_PacketEndpointCtrl->SetOnClosedCallback([this](){ OnClosed(); });
312  m_PacketEndpointCtrl->Start();
313  m_PacketEndpointCtrl->Send(HdlcdSessionHeader::Create(HdlcdSessionDescriptor(SESSION_TYPE_TRX_STATUS, SESSION_FLAGS_NONE), m_SerialPortName));
314  m_OnConnectedCallback(true);
315  return;
316  } // if
317 
318  if ((m_eTcpSocketDataState == SOCKET_STATE_CONNECTING) || (m_eTcpSocketCtrlState == SOCKET_STATE_CONNECTING)) {
319  // Needs to wait for the second socket
320  return;
321  } // if
322 
323 
324  if (m_eTcpSocketDataState == SOCKET_STATE_CONNECTED) {
325  // The control socket failed after the data socket succeeded
326  assert(m_eTcpSocketCtrlState == SOCKET_STATE_ERROR);
327  m_eTcpSocketDataState = SOCKET_STATE_ERROR;
328  m_TcpSocketData.close();
329  m_OnConnectedCallback(false);
330  return;
331  } // else if
332 
333  if (m_eTcpSocketCtrlState == SOCKET_STATE_CONNECTED) {
334  // The data socket failed after the control socket succeeded
335  assert(m_eTcpSocketDataState == SOCKET_STATE_ERROR);
336  m_eTcpSocketCtrlState = SOCKET_STATE_ERROR;
337  m_TcpSocketCtrl.close();
338  m_OnConnectedCallback(false);
339  return;
340  } // else if
341 
342  // Both sockets failed
343  assert(m_eTcpSocketDataState == SOCKET_STATE_ERROR);
344  assert(m_eTcpSocketCtrlState == SOCKET_STATE_ERROR);
345  m_OnConnectedCallback(false);
346  return;
347  }
